
usb start
hostname=baylibre-acme
device=eth0
autoconf=on

loadaddr=0x82000000
fdtaddr=0x88000000
rdaddr=0x88080000

initrd_high=0xffffffff
fdt_high=0xffffffff

mmcroot=/dev/mmcblk0p2

## yocto name
#
fdtfile=zImage-am335x-boneblack.dtb

loadximage=load mmc 0:1 ${loadaddr} zImage
loadxfdt=load mmc 0:1 ${fdtaddr} ${fdtfile}

loadall=run loadximage; run loadxfdt;

mmcargs=setenv bootargs console=tty0 ip=dhcp console=${console} ${optargs} ${cape_disable} ${cape_enable} root=${mmcroot} rootfstype=${mmcrootfstype} ${cmdline}

uenvcmd=run loadall; run mmcargs; bootz ${loadaddr} - ${fdtaddr};
